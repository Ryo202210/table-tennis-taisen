    <script>

        let players = [];

        let scores = {};


        window.onload = () => {

            const savedPlayers = localStorage.getItem('pingpong_players_pop');

            const savedScores = localStorage.getItem('pingpong_scores_pop');


            if (savedPlayers) {

                players = JSON.parse(savedPlayers);

                scores = JSON.parse(savedScores || '{}');

                renderPlayerInputs();

                if (players.length >= 4) {

                    generateTable(false);

                }

            } else {

                // 初期の名前を「プレイヤー1, 2...」に変更

                players = ['プレイヤー1', 'プレイヤー2', 'プレイヤー3', 'プレイヤー4'];

                renderPlayerInputs();

            }

        };


        function renderPlayerInputs() {

            const container = document.getElementById('player-inputs');

            container.innerHTML = '';

            players.forEach((name, index) => {

                const div = document.createElement('div');

                div.className = 'flex items-center bg-orange-50 p-3 rounded-2xl border-2 border-white shadow-sm';

                div.innerHTML = `

                    <input type="text" value="${name}" 

                        onchange="updatePlayerName(${index}, this.value)"

                        class="bg-transparent border-none focus:ring-0 text-sm w-full font-black text-orange-700"

                        placeholder="名前">

                    <button onclick="removePlayer(${index})" class="text-orange-200 hover:text-red-500 font-black px-2 text-xl">&times;</button>

                `;

                container.appendChild(div);

            });

            document.getElementById('add-btn').style.display = players.length >= 9 ? 'none' : 'inline-block';

        }


        function addPlayer() {

            if (players.length < 9) {

                // 追加される時も「プレイヤー○」の形式にする

                players.push(`プレイヤー${players.length + 1}`);

                renderPlayerInputs();

                saveData();

            }

        }


        function removePlayer(index) {

            if (players.length > 4) {

                players.splice(index, 1);

                renderPlayerInputs();

                saveData();

            } else {

                showToast("最低4人必要だよ！");

            }

        }


        function updatePlayerName(index, name) {

            players[index] = name;

            saveData();

        }


        function generateTable(withToast = true) {

            if (players.length < 4) {

                showToast("4人以上で遊ぼう！");

                return;

            }


            document.getElementById('results-section').classList.remove('hidden');

            document.getElementById('ranking-section').classList.remove('hidden');


            const header = document.getElementById('league-header');

            header.innerHTML = '<th class="p-2 text-orange-300 text-xs">対戦</th>';

            players.forEach(p => {

                header.innerHTML += `<th class="p-2 text-blue-600 font-black truncate max-w-[60px]">${p}</th>`;

            });


            const body = document.getElementById('league-body');

            body.innerHTML = '';

            players.forEach((p1, i) => {

                let row = `<tr><td class="p-2 font-black text-blue-600 bg-blue-50 rounded-xl mb-1">${p1}</td>`;

                players.forEach((p2, j) => {

                    if (i === j) {

                        row += `<td class="bg-slate-100 rounded-xl text-center text-slate-300">/</td>`;

                    } else {

                        const key = i < j ? `${i}-${j}` : `${j}-${i}`;

                        const score = scores[key] || { s1: '', s2: '' };

                        const val1 = i < j ? score.s1 : score.s2;

                        const val2 = i < j ? score.s2 : score.s1;


                        row += `

                            <td class="p-1 text-center">

                                <div class="flex flex-col sm:flex-row items-center justify-center gap-1 bg-white border-2 border-slate-50 rounded-xl p-1 shadow-sm">

                                    <input type="number" value="${val1}" min="0"

                                        onchange="updateScore(${i}, ${j}, 1, this.value)"

                                        class="cell-input text-blue-500 font-black">

                                    <span class="text-slate-300 text-[10px]">vs</span>

                                    <input type="number" value="${val2}" min="0"

                                        onchange="updateScore(${i}, ${j}, 2, this.value)"

                                        class="cell-input text-red-400 font-black">

                                </div>

                            </td>`;

                    }

                });

                row += '</tr>';

                body.innerHTML += row;

            });


            renderMatchList();

            updateRanking();

            if(withToast) showToast("対戦表をつくったよ！");

        }


        function renderMatchList() {

            const listContainer = document.getElementById('match-list');

            listContainer.innerHTML = '';

            

            let matches = [];

            for (let i = 0; i < players.length; i++) {

                for (let j = i + 1; j < players.length; j++) {

                    matches.push({ p1: i, p2: j });

                }

            }


            matches.forEach((match, idx) => {

                const key = `${match.p1}-${match.p2}`;

                const isDone = scores[key] && scores[key].s1 !== '' && scores[key].s2 !== '';

                const scoreDisplay = isDone ? `${scores[key].s1} - ${scores[key].s2}` : '未対戦';

                

                const div = document.createElement('div');

                div.className = `p-3 rounded-xl border-2 flex justify-between items-center ${isDone ? 'bg-white border-green-200 text-green-700' : 'bg-white border-blue-100 text-blue-400'}`;

                div.innerHTML = `

                    <div class="flex items-center">

                        <span class="bg-slate-100 text-slate-400 rounded-lg w-6 h-6 flex items-center justify-center text-[10px] mr-2">${idx + 1}</span>

                        <span class="truncate max-w-[120px]">${players[match.p1]} vs ${players[match.p2]}</span>

                    </div>

                    <span class="text-xs font-black px-2 py-1 rounded-lg ${isDone ? 'bg-green-100' : 'bg-slate-100'}">${scoreDisplay}</span>

                `;

                listContainer.appendChild(div);

            });

        }


        function updateScore(i, j, pos, val) {

            const p1Idx = i < j ? i : j;

            const p2Idx = i < j ? j : i;

            const key = `${p1Idx}-${p2Idx}`;

            

            if (!scores[key]) scores[key] = { s1: '', s2: '' };

            

            if (i < j) {

                if (pos === 1) scores[key].s1 = val;

                else scores[key].s2 = val;

            } else {

                if (pos === 1) scores[key].s2 = val;

                else scores[key].s1 = val;

            }


            saveData();

            updateRanking();

            renderMatchList();

        }


        function updateRanking() {

            let stats = players.map((name, index) => ({

                id: index,

                name: name,

                wins: 0,

                losses: 0,

                setsWon: 0,

                setsLost: 0

            }));


            Object.keys(scores).forEach(key => {

                const [idx1, idx2] = key.split('-').map(Number);

                const s1 = parseInt(scores[key].s1);

                const s2 = parseInt(scores[key].s2);


                if (!isNaN(s1) && !isNaN(s2)) {

                    stats[idx1].setsWon += s1;

                    stats[idx1].setsLost += s2;

                    stats[idx2].setsWon += s2;

                    stats[idx2].setsLost += s1;


                    if (s1 > s2) {

                        stats[idx1].wins++;

                        stats[idx2].losses++;

                    } else if (s2 > s1) {

                        stats[idx2].wins++;

                        stats[idx1].losses++;

                    }

                }

            });


            stats.sort((a, b) => {

                if (b.wins !== a.wins) return b.wins - a.wins;

                return (b.setsWon - b.setsLost) - (a.setsWon - a.setsLost);

            });


            const rankingBody = document.getElementById('ranking-body');

            rankingBody.innerHTML = '';

            stats.forEach((stat, index) => {

                const diff = stat.setsWon - stat.setsLost;

                const emoji = index === 0 ? '👑' : index === 1 ? '🥈' : index === 2 ? '🥉' : '';

                

                rankingBody.innerHTML += `

                    <tr class="border-b border-slate-50 hover:bg-green-50 transition">

                        <td class="py-4 px-2 font-black text-lg text-slate-400">

                            ${emoji ? emoji : index + 1}

                        </td>

                        <td class="py-4 px-2 font-black text-slate-700">${stat.name}</td>

                        <td class="py-4 px-2">

                            <span class="text-orange-500 font-black text-lg">${stat.wins}</span><span class="text-xs text-slate-400 ml-1">勝</span>

                            <span class="text-slate-300 font-black ml-1">${stat.losses}敗</span>

                        </td>

                        <td class="py-4 px-2 text-right">

                            <div class="inline-flex flex-col items-end">

                                <span class="font-black ${diff >= 0 ? 'text-green-500' : 'text-red-400'}">${diff > 0 ? '+' : ''}${diff}</span>

                                <span class="text-[10px] text-slate-300 font-bold">set: ${stat.setsWon}/${stat.setsLost}</span>

                            </div>

                        </td>

                    </tr>

                `;

            });

        }


        function saveData() {

            localStorage.setItem('pingpong_players_pop', JSON.stringify(players));

            localStorage.setItem('pingpong_scores_pop', JSON.stringify(scores));

        }


        function confirmReset() {

            if (confirm("データをぜんぶ消して、最初からやりなおす？")) {

                localStorage.removeItem('pingpong_players_pop');

                localStorage.removeItem('pingpong_scores_pop');

                location.reload();

            }

        }


        function showToast(msg) {

            const toast = document.getElementById('toast');

            toast.innerText = msg;

            toast.classList.remove('opacity-0');

            toast.classList.add('opacity-100', '-translate-y-2');

            setTimeout(() => {

                toast.classList.add('opacity-0');

                toast.classList.remove('opacity-100', '-translate-y-2');

            }, 2000);

        }
